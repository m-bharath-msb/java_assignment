public class Print2DArray {
    public static void main(String[] args) {
        int marks1[][] = {
            {1, 4, 2},
            {9, 8, 7},
            {2, 4, 5}
        };
        
        for (int i = 0; i < marks1.length; i++) {
            for (int j = 0; j < marks1[i].length; j++) {
                System.out.print(marks1[i][j] + " ");
            }
            System.out.println();
        }
    }
}
